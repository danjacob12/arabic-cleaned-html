أوال)

السـارية ذات الصلة باإليرادات ومنها ما ورد في ملحق رقم )1( .

ثالثــا)

 التأكيــد علــى مــا قضــى بــه الفصــل الثالــث (تحصيــل اإليــرادات) مــن نظــام إيــرادات الدولــة وعلــى الجهــة التــي ترتبــط بنظــام ســداد التنســيق مــع وزارة الماليــة لربطهــا بنظــام تحصيــل (النظــام المركــزي فــي وزارة الماليـة لربـط الجهـات الحكوميـة بنظـام سـداد للمدفوعـات وبأنظمـة الدفـع األخـرى) وتحصيـل كافـة إيراداتهـا

عبــر هــذا النظــام مــع إيقــاف جميــع أســاليب التحصيــل األخــرى.

رابعــا)

المـادة (السادسـة عشـرة) مـن الائحـة التنفيذيـة لنظـام إيـرادات الدولـة الصـادرة بالقـرار الـوزاري رقـم )860( وتاريـخ 1432/3/13هــ المعدلـة بالقـرار الـوزاري رقـم )901( وتاريـخ 1439/2/24هــ، علـى أن تكـون مصنفـة بنــاء علــى التصنيــف االقتصــادي وفقــا لدليــل إحصــاءات ماليــة الحكومــة 2014(م)، وصــرف مــا يقابــل تلــك اإليـرادات المقـدرة مـن االعتمـادات المبلغـة لكـم عبـر منصـة اعتمـاد وفقـا لما ورد بتعليمـات تنفيذ الميزانية

العامـة للدولـة والتعليمـات الماليـة والمحاسـبية.

 عنــد اعــداد أمــر الدفــع الخــاص بالصــرف مــن اعتمــاد مقابــل اإليــراد، يــدون

علــى أمــر الدفــع فــي خانــة

(وذلــك مقابــل ).. عبــارة (مخصــص مــن اإليــراد.)

الحســابات المشــار إليهــا فــي الفقــرة )2( أعــاه، والبــدء بإيــداع إيــرادات كافــة األنشـــطة لعــام 2022(م)

بحســاب جــاري وزارة الماليــة وفقــا لمــا ورد فــي هــذه التعليمــات .

 علـى الجهـات التـي ينـص تنظيمهـا علـى إبـرام عقـود اسـتثمارية االسـتمرار علـى تلـك العقـود والتنسـيق حيــال ذلــك مــع الهيئــة العامــة لعقــارات الدولــة فيمــا يخــص اســتثمار األراضــي والمواقــع وعلــى الجهــات

إيــداع العوائــد أو األربــاح الناجمــة عــن تلــك العقــود لحســاب جــاري وزارة الماليــة بالبنــك المركــزي الســعودي.

خامسا) تعليمات عامة:

.1 

بعـد توريـد اإليـرادات المباشـرة لحسـاب جـاري وزارة الماليـة بالبنـك المركـزي السـعودي يتـم إجـراء القيـود المحاسـبية وتسـجيلها بالدفاتـر الرسـمية كذلـك مـا يتـم صرفـه مـن اعتمـاد مقابـل إيـراد وإظهـار ذلـك ضمـن

الحسـاب الشـهري والحسـاب الختامـي للجهـة.

.2 

تقـدم الجهــة للـوزارة إيراداتهــا المقـدرة لـكل سـنة ماليـة وفقـا لخطـة مشـروع الميزانيـة والتـي تنتهجهـا

الــوزارة ويتــم تعميمهــا مــع بدايــة المشــروع، وتشــمل التقديــرات المقدمــة مــن الجهـــة علــى اآلتــي:

أ) بيان باإليرادات النقدية والمستقطعة من المنبع . ب) اإلحصائيـات المؤيـدة الحتسـاب التقديـرات لـكل حسـاب مـن حسـابات اإليـرادات لعاميـن ماليين سـابقين

علـى األقـل .

ج) العوامل المؤثرة التي أخذ بهـا عند إعداد التقديرات .

د) مبررات الزيادة أو النقص في التقديرات .

هـ) بيان بالمحصل الفعلي خال ثاث سنوات مالية سابقة مفص ًا لكل سنة ولكل إيراد تم تحصيله.

.3 

للوزارة دعوة المسؤولين والمختصين في الجهــة لمناقشتهــم حول التقديرات المقدمة، ولهــا تعديلهــا

وإبـاغ الجهــة بمـا يتـم اعتمـاده مـن تقديـرات .

.4 
اعتمـاد اللوائـح والتعليمـات واألنظمـة المعتمـدة نظامـا لـدى الجهـات بمـا اليتعـارض مـع األمـر الملكـي )55685( وتاريــخ 1438/11/30هــ ومرســوم الميزانيــة العامــة للدولــة رقــم (م)39/ وتاريــخ 1443/5/8هــ

وهــذه التعليمــات.

.5 

مع عدم اإلخال بما ورد في األنظمة األخرى، ال يحق للجهـة استثمار إيراداتهـا النقدية . .

6 

علـى الجهــات الحكوميـة قيـد المبالـغ التـي تصـرف زيـادة عـن المسـتحق ويتـم تحصيلهــا فـي سـنة ماليـة

الحقـة إلـى حسـاب اإليـرادات مـا عـدا المبالـغ التـي تخـص مشـاريع قائمـة فتسـتبعد مـن بنودهـا.

.7

علــى الجهـــة قيــد اإليــرادات التــي تحصــل فــي بدايــة الســنة الماليــة فــي اليــوم الخامــس عشــر فأكثــر مــن الشهــر ضمـن إيـرادات الشهــر الـذي يليـه، ومـا يتـم تحصيلـه فـي نهــاية السـنة الماليـة فـي اليـوم الرابـع عشـر

فأقـل مـن الشهــر ضمـن إيـرادات الشهــر السـابق لـه .

.8

علــى الجهـــات الحكوميــة اســتخدام أنظمــة الحاســب اآللــي فــي جميــع العمليــات الماليــة والمحاســبية، والتحـول مـن الوسـائل التقليديـة فـي مسـك السـجات وإعـداد الحسـابات والبيانـات الماليـة إلـى الوسـائل

اإللكترونيــة .

.9

ال يجــوز للجهـــة اســتبعاد مبلــغ قيــد لحســاب اإليــرادات إال بموافقــة وزارة الماليــة أو المراقــب المالــي

بموجــب الصاحيــات المخولــة لــه بالائحــة .

.10

 ال يتم استبعاد مبلغ من حساب اإليرادات إال من الحساب الذي قيد فيه.

.11 

تتولــى وزارة الماليــة إجــازة طلبــات اســتبعاد المبالــغ مــن حســاب اإليــرادات الخاصــة بالمؤسســات والشــركات بعــد إرفــاق الجهـــة المســتندات المطلوبــة وفقــا للمــادة )38( مــن الائحــة التنفيذيــة لنظــام

إيــرادات الدولــة .

أن يؤشـر أمـام المبلـغ الـذي صـرف برقـم وتاريـخ مسـتند الصـرف .

الماليــة بالبيانــات اآلتيــة:

أ) اإليرادات التي تم قيدهـا وتسجيلهـا بالدفاتر.

ب) المبالغ التي تم استبعادهـا من حساب اإليرادات بموافقة وزارة المالية عليهـا.

ج) المبالغ التي تم إجازة استبعادهـا من قبل المراقب المالي بحسب الصاحية الممنوحة له.

د) تســجيل حقــوق الخزينــة المســتحقة علــى المديــن  بنظــام المديونيــات علــى منصــة اعتمــاد موضحــا فيهــا اســم المديــن (شــركة، مؤسســة، فــرد) ومبلــغ المديونيــة ورقــم الســجل التجــاري أو رقــم الهويــة

الوطنيــة أو اإلقامــة وتاريــخ المديونيــة ونوعهــا والمبلــغ المســدد.

.1  تــم تقديــر اعتمــادات الرواتــب والبــدالت وأجــور العمــال ورواتــب الوظائــف المؤقتــة المدرجــة فــي النفقــات العامـة (تعويضـات العامليـن) وكذلـك رواتـب الوظائـف المدرجـة فـي النفقـات علـى البرامـج لمـدة إثنـي عشـر شـهرا فـي ميزانيـة الجهـات الحكوميـة للعـام المالـي علـى أن يتـم الصرف فقط للموظفيـن المعينين والمتعاقد

ثالثــا:

 عــدم تجــاوز أعــداد الوظائــف الثابتــة، والوظائــف المؤقتــة، والوظائــف المدرجــة ضمــن النفقــات علــى

البرامــج، وأن ال يتــم االلتــزام بمــا يزيــد علــى االعتمــادات المخصصــة لهــذه الوظائــف.

رابعـا:

ينبغـي عنـد صـرف البـدل المقـر للعامليـن الذيـن يعملـون بالمناطـق النائيـة والحدوديـة مراعـاة مقتضى الموافقـة السـامية المبلغـة بخطـاب ديـوان رئاسـة مجلـس الـوزراء رقـم /2262/7(ر) وتاريـخ 1409/2/23هــ القاضــي بتحديــد البــدل بنســبة )%5( و )%20( كحــد أعلــى والقــرارات الصــادرة مــن وزارة المــوارد البشــرية والتنميـة االجتماعيـة بشـأن أسـماء القـرى والهجـر والمراكـز التـي يتـم دراسـتها وبـدل المناطـق النائيـة المقـرر

لهـا وفـق النسـب الموضحـة أمـام كل منهـا.

خامسـا:

اسـتنادا لنظـام الوظائـف المؤقتـة الصـادر بالمرسـوم الملكـي رقـم (م)30/ وتاريـخ 1385/9/12هــ يراعــى تطبيــق مــا تضمنتــه المــواد 1(،2،)3 علــى المعينيــن علــى الوظائــف المؤقتــة المدرجــة أعدادهــا فــي كشــوف الميزانيــة مــع أهميــة التقيــد باألعــداد واالعتمــادات المدرجــة فــي الميزانيــة وعــدم التعييــن إال بالتوافـق مـع وزارتـي الماليـة والمـوارد البشـرية والتنميـة االجتماعيـة وبمـا ال يتجـاوز تلـك األعـداد بـأي حـال مــن األحــوال, واألمــر الســامي رقــم /1895(م ب) وتاريــخ 1432/3/23هــ وتعميــم وزارة المــوارد البشــرية والتنميـة االجتماعيـة رقـم )495( وتاريـخ 1432/4/28هــ وسـيكون مديـر الشـؤون اإلداريـة والماليـة ومديـر

شــؤون الموظفيــن مســؤولين عــن أي تجــاوز.

.2

 االعتمــادات المدرجــة لنفقــات أجــور العمــال مخصصــة لصــرف األجــور الشــهرية فقــط, كمــا يراعــى عــدم تكليــف العمــال بأعمــال إداريــة أو توظيــف عمــال مؤقتيــن أو انتدابهــم لمهــام خــارج مقــر أعمالهــم أو إلغــاء

إجازاتهــم وصــرف تعويــض عــن ذلــك.

.3  

عـدم التوسـع فـي تكليـف العمـال وشـاغلي الوظائـف المؤقتـة بأعمـال إضافيـة خـال األيـام العاديـة وأيـام

العطـل الرسـمية. وإن تطلـب األمـر ذلـك، يتـم األخـذ بأسـلوب العمـل علـى شـكل ورديـات خـال األيـام العاديـة، وأســلوب المناوبــة خــال العطــل الرســمية بمــا يحقــق أداء العمـــل المطلــوب دون أن يترتــب علــى ذلــك مبالــغ

إضافية.

.4  

االلتـزام باإلجـراءات والسياسـات المرتبطـة بالتوظيـف واإلحـال والتوطيـن بحسـب قـرار مجلـس الـوزراء رقـم )20( وتاريــخ 1440/1/8هــ المبلــغ ببرقيــة الديــوان الملكــي رقــم )2150( وتاريخ1440/1/11هــ واســتكمال اإلجــراءات النظاميــة بشــأنها، واالتفــاق مــع الجهــة المشــار إليهــا فــي البنــد (ثالثــا) مــن قــرار مجلــس الــوزراء عنــد

الرغبــة فــي إيجــاد بوابــات للتوظيــف وقواعــد للبيانــات.

.5

 االلتــزام بقــرار مجلــس الــوزراء رقــم )337( وتاريــخ 1439/6/25هــ بخصــوص توطيــن شــغل الوظائــف فــي عقــود التشــغيل والصيانــة، وفقــا لنســب التوطيــن الــواردة فــي قــرار معالــي وزيــر العمــل والتنميــة االجتماعيــة رقـم )35490( وتاريـخ 1441/2/21هــ المبنـي علـى قـرار مجلـس الـوزراء المشـار إليـه، والمتضمـن إلـزام الـوزارات واألجهــزة الحكوميــة األخــرى بتطبيــق تلــك النســب علــى جميــع عقــود التشــغيل والصيانــة. وعلــى الــوزارات واألجهـزة الحكوميـة األخـرى االلتـزام بذلـك عنـد إعـداد الشـروط والمواصفـات، وتوصيـف الوظائـف ومتطلبات

شــغلها فــي العقــود، وذلــك تنفيــذا لألوامــر والتوجيهــات الســامية بشــأن توطينهــا.

.6

االلتــزام بدليــل توطيــن عقــود التشــغيل والصيانــة فــي الجهــات العامــة الصــادر مــن وزارة المــوارد البشــرية والتنميــة االجتماعيــة بنــاء علــى توجيــه مجلــس الــوزراء المعمــم ببرقيــة الديــوان الملكــي رقــم )21567( وتاريــخ 1441/4/1هــ، المتضمـن منـح معالـي وزيـر المـوارد البشـرية والتنميـة االجتماعيـة صاحيـة إصـدار دليـل توطيـن عقــود التشــغيل والصيانــة فــي الجهــات العامــة، وتعميمــه عليهــا، وإلزامهــا بتطبيقــه علــى عقــود التشــغيل

والصيانــة، علــى أن ال تتجــاوز التكاليــف المترتبــة علــى ذلــك مــا تــم اعتمــاده بالميزانيــة لهــذا الغــرض.

.7

عــدم التعييــن علــى الوظائــف الشــاغرة إال بالتوافــق مــع وزارتــي الماليــة والمــوارد البشــرية والتنميــة االجتماعيــة، ويســتثنى مــن ذلــك الجهــات المشــمولة بالتعييــن وفقــا لقــرار مجلــس الــوزراء رقــم )20( وتاريــخ 1440/1/8 هــ علــى أن ال يــؤدي ذلــك لتجــاوز االعتمــادات المخصصــة فــي ميزانيــة الجهــة. وســيكون مديــر

الشــؤون اإلداريــة والماليــة ومديــر شــؤون الموظفيــن مســؤولين عــن أي تجــاوز.

.8

يجـب علـى الجهـات الحكوميـة عنـد رفعهـا طلبـات لتعديـل الوظائـف ضمـن مشـروع الميزانيـة القادمـة عـدم

اشـغال هـذه الوظائـف لحيـن صـدور وتبليـغ مرسـوم الميزانية.

.9  

ال يجــوز تعييــن أو ترقيــة الموظفيــن والمســتخدمين والعمــال إال علــى الوظائــف والمبالــغ المعتمــدة فــي

الميزانيــة العامــة للدولــة وفقــا لإلجـراءات النظاميــة.

.10

 تنفيذا لمقتضى البند الثاني عشر من مرسوم الميزانية العامة للدولة رقم (م)39/ وتاريخ 1443/5/8هـ،

يتـم التقيـد بمـا يلي:

أو ًال:

 تعتمـد التشـكيات اإلداريـة (الوظيفيـة) لـكل جهـة بحسـب مـا صـدرت بـه الميزانيـة العامـة للدولـة، وال

يجـوز تعديلهـا إال وفقـا لإلجـراءات النظاميـة المتبعـة.

ثانيـا: 

ال يجـوز خـال السـنة الماليـة إحـداث أو رفـع وظائـف أو مراتـب أو ُر َتـب خـاف مـا صـدرت بهـا الميزانيـة

العامــة للدولــة، وي ْســت ْث َنى مــن ذلــك تعييــن الــوزراء، ومــن فــي مرتبــة وزيــر والمرتبــة الممتــازة ومــا يعادلهــا،

والوظائــف التــي ُت ْحــدث وفقــا للشــروط التــي تضمنهــا نظــام الوظائــف المؤقتــة.

ثال ًثــــــــــــــــا:

 يجــــــــــــــــوز بقــــــــــــــــرار مــــــــــــــــن وزيــــــــــــــــر المــــــــــــــــوارد البشــــــــــــــــرية والتنميــــــــــــــــة االجتماعيــــــــــــــــة تخفيــــــــــــــــض المراتــــــــــــــــب

أو المســــــــــــــــتويات، أو تحويــــــــــــــــر مســــــــــــــــميات وفقــــــــــــــــ ًا لمقتضيــــــــــــــــات قواعــــــــــــــــد تصنيــــــــــــــــف الوظائــــــــــــــــف،

وذلــك بنــاء علــى توصيــة مــن لجنــة مكونــة مــن ممثليــن مــن وزارة الماليــة ووزارة المــوارد البشــرية والتنميــة االجتماعيــة والجهــة ذات العاقــة، علــى أنــه لوزيــر المــوارد البشــرية والتنميــة االجتماعيــة منــح صاحيــات

التحويــر و التخفيــض للوزيــر المختــص وفقــا لضوابــط وإجــراءات تحويــر وتخفيــض الوظائــف، وهــي كالتالــي:

أ- الضوابط التنظيمية والتصنيفية: تكــون الموافقــة علــى تحويــر وتخفيــض الوظائــف بقــرار مــن الوزيــر المختــص أو رئيــس الجهــة المســتقلة

وفقــا لمــا يلــي:

.1

  أن يكـون التحويـر أو التخفيـض وفقـا لمتطلبـات وحاجـة العمـل الفعليـة التـي ال تقبـل التأجيـل إلـى

موعـد صـدور الميزانيـة العامـة للدولـة.

.2

أن تكــون أولويــة التحويــر إلــى مــا يخــدم النشــاط األساســي للجهــة وأن تراعــي قواعــد تصنيــف

الوظائــف ومهمــات ومســؤوليات الوحــدة اإلداريــة.

.3

أن يقتصــر التحويــر علــى الوظائــف فــي الســلم الوظيفــي الواحــد، وعــدم جــواز التحويــر مــن ســلم

رواتــب إلــى ســلم رواتــب آخــر.

.4  

مراعــاة التــوازن الوظيفــي فــي الوحــدات اإلداريــة للجهــة وفقــا لحجــم العمــل ومســتوياته وعــدم

تركيــز الوظائــف فــي إدارات أو وحــدات علــى حســاب الوحــدات األخــرى.

.5

أن يتفق مسمى الوظيفة المقترح مع مهام وواجبات الوحدة اإلدارية التابعة لها.

.6 أن ال يفوق مستوى الوظيفة المقترحة بالتحوير مستوى الوظيفة اإلشرافية لوحدتها. .7  أن ال يــؤدي التحويــر أو التخفيــض إلــى إلغــاء مركــز إشــرافي قائــم فــي الوحــدة اإلداريــة أو إيجــاد

وحــدة إداريــة جديــدة غيــر معتمــدة بالهيــكل التنظيمــي الرســمي.

.8

عــدم تحويــر الوظائــف اإلشــرافية الهيكليــة المحــددة بالهيــكل التنظيمــي للجهــة إال فــي حالــة االزدواجيــة بوجــود مســمى مماثــل يقــوم بالمهــام نفســها، أو وجــود تعديــل فــي الهيــكل التنظيمــي

للجهــة إذا تــم اعتمــاده.

.9

 يقتصر التحوير والتخفيض على الوظائف الشـاغرة وفقا لسـجات وزارة الموارد البشـرية والتنمية االجتماعيـة (الوظيفـة العامـة)، ويجـوز تحويـر الوظائـف المشـغولة وذلـك فـي حالـة وجـود متطلبـات تنظيميــة وتصنيفيــة أو أن حاجــة العمــل تســتدعي ذلــك بشــرط توفــر متطلبــات شــغل الوظيفــة

بالمســمى المقتــرح لــدى شــاغلها والقواعــد التنظيميــة األخــرى وفقــا لدليــل تصنيــف الوظائــف.

.10

يشــترط الحصــول علــى إقــرار خطــي بالموافقــة مــن الموظــف شــاغل الوظيفــة فــي حــال كــون الوظيفــة المــراد تحويرهــا مشــغولة ويترتــب علــى التحويــر إلغــاء بــدل أو مكافــأة أو نقــل إلــى مدينــة

أخــرى أو خفــض المرتبــة (فــي حــال إعــادة تقييمــه مــن الوظائــف المميــزة إلــى غيرهــا.)

.11

 عــدم تحويــر الوظائــف الفنيــة أو الحرفيــة المشــغولة إلــى وظائــف إداريــة أو إداريــة مســاعدة إال بموجـب تقريـر طبـي معتمـد مـن الهيئـة الطبيـة يوصـي بتغييـر مجـال عمـل شـاغلها إلـى عمـل إداري

لظروفــه الصحيــة.

.12

لجنــة مكونــة مــن ممثليــن مــن: وزارة الماليــة ووزارة المــوارد البشــرية والتنميــة االجتماعيــة.

العاشـر مـن السـنة الماليـة.

خامســا: 

وفقــا للبنــد الثانــي عشــرمن مرســوم الميزانيــة العامــة للدولــة رقــم (م)39/ وتاريــخ 1443/5/8هــ، يجــوز بقــرار مــن وزيــر الجهــة للجهــات المرتبطــة والتابعــة لــه أو رئيــس الجهــة المنقــول منهــا وموافقــة وزيــر الجهــة للجهــات المرتبطــة والتابعــة لــه أو رئيــس الجهــة المنقــول إليهــا نقــل الوظائــف المعتمــدة للمرتبــة (العاشـرة) فمـادون خـارج الجهـة، وذلـك بعـد االعتمـاد مـن اللجنـة المشـكلة مـن ممثليـن مـن وزارة الماليـة ووزارة المــوارد البشــرية والتنميــة االجتماعيــة (مــع ضــرورة االلتــزام بتزويــد نســخة مــن القــرار الــوزاري لــوزارة

الماليــة خــال أســبوع مــن تاريــخ صــدور القــرار)

سادســا: 

يجــوز للوزيــر المختــص أو رئيــس الجهــة المســتقلة نقــل الوظائــف مــن المرتبــة (العاشــرة) فمــا دون داخــل التشــكيل اإلداري للجهــاز الواحــد مــع االلتــزام بالضوابــط والمعاييــر التــي حددهــا األمــر الســامي رقــم

)63336( وتاريــخ 1439/12/8هــ، وهــي كمــا يلــي:

• أن تكون الموافقة على نقل الوظيفة من قبل الوزير المختص أو رئيس الجهة المستقلة.

•  أن ال يؤدي النقل إلى إلغاء مركز إشرافي قائم في الوحدة اإلدارية.

• أن ال يؤدي النقل إلى إيجاد وحدة إدارية جديدة غير معتمدة في الهيكل التنظيمي.

• أن يراعى عند النقل قواعد التصنيف ومهمات ومسؤوليات الوحدة اإلدارية المنقول إليها.

•  أن يتناســب النقــل مــع األعــداد واألســماء والمهمــات المتعلقــة بالوحــدة اإلداريــة المنقــول إليهــا

الوظيفــة.

• أن ال يؤدي نقل الوظيفة إلى انقطاع السلسلة الوظيفية للوحدة اإلدارية.

•  تزويد وزارة المالية بصورة من قرار النقل – عبر أنظمة الوزارة اإللكترونية-لتحديث بيانات الوظيفة.

• تزويــد وزارة المــوارد البشــرية والتنميــة االجتماعيــة بصــورة مــن قــرار النقــل – عبــر أنظمــة الــوزارة
